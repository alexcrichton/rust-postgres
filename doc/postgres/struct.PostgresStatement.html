fn execute(&self, params: &[&ToSql]) -> PostgresResult<uint>

Executes the prepared statement, returning the number of rows modified.

If the statement does not modify any rows (e.g. SELECT), 0 is returned.

Example

let stmt = conn.prepare("UPDATE foo SET bar = $1 WHERE baz = $2").unwrap();
match stmt.execute([&bar as &ToSql, &baz as &ToSql]) {
    Ok(count) => println!("{} row(s) updated", count),
    Err(err) => println!("Error executing query: {}", err)
}

fn query<'a>(&'a self, params: &[&ToSql]) -> PostgresResult<PostgresRows<'a>>

Executes the prepared statement, returning an iterator over the resulting rows.

Example

let stmt = conn.prepare("SELECT foo FROM bar WHERE baz = $1").unwrap();
let mut rows = match stmt.query([&baz as &ToSql]) {
    Ok(rows) => rows,
    Err(err) => fail!("Error running query: {}", err)
};
for row in rows {
    let foo: i32 = row["foo"];
    println!("foo: {}", foo);
}
